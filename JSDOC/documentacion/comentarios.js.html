/**
 * 
 * @param {number[]} datos - Matriz de números a procesar
 * @param {CallbackParaProcesar} funcion - Función a ejecutar tras procesar los números 
 */
function procesarDatos(datos, funcion){
    const resultado=datos.map(x=>x*2);
    funcion(resultado);
}   

/**

 */
